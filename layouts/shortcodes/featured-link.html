{{/* Featured link — an attractive "link preview" card for an external post. It fetches the target page at build time and pulls its Open Graph metadata (og:image, og:title, og:description, og:site_name). Anything it can't find — or anything you want to override — can be passed explicitly as a param, so the card still renders if the remote site is unreachable at build time. Usage: {{< featured-link url="https://dewofyouryouth.com/post/my-arabic-music-liner-notes/" >}} Params: url (required) the page to feature. title (optional) overrides the fetched og:title. description (optional) overrides the fetched og:description. image (optional) overrides the fetched og:image. site (optional) small kicker label; defaults to the URL's domain. label (optional) eyebrow text; defaults bilingually ("from my other blog"). cta (optional) call-to-action text; defaults bilingually ("Read the post"). dir (optional) text direction of the card body, "ltr" (default) or "rtl". */}} {{- $url := .Get "url" -}} {{- if not $url -}} {{- errorf "featured-link: a url is required (in %s)" .Page.File.Path -}} {{- end -}} {{- $isAr := or (eq (.Page.Params.postLang | string) "ar") (eq .Page.Language.Lang "ar") -}} {{- $dir := .Get "dir" | default "ltr" -}} {{- $label := .Get "label" | default (cond $isAr "من مدوّنتي التانية" "from my other blog") -}} {{- $cta := .Get "cta" | default (cond $isAr "اقرا البوست" "Read the post") -}} {{- $title := .Get "title" -}} {{- $desc := .Get "description" -}} {{- $img := .Get "image" -}} {{- $site := .Get "site" -}} {{/* --- pull Open Graph metadata from the target page at build time --- */}} {{- $html := "" -}} {{- with try (resources.GetRemote $url) -}} {{- with .Err -}} {{- warnf "featured-link: could not fetch %s: %s" $url . -}} {{- else with .Value -}} {{- $html = .Content -}} {{- end -}} {{- end -}} {{- if $html -}} {{- range $field := slice "og:image" "og:title" "og:description" "og:site_name" -}} {{- $val := "" -}} {{- $p1 := findRESubmatch (printf `(?is)]+(?:property|name)=["']%s["'][^>]*?content=["']([^"']*)["']` $field) $html 1 -}} {{- with $p1 -}}{{- $val = index (index . 0) 1 -}}{{- end -}} {{- if not $val -}} {{- $p2 := findRESubmatch (printf `(?is)]+content=["']([^"']*)["'][^>]*?(?:property|name)=["']%s["']` $field) $html 1 -}} {{- with $p2 -}}{{- $val = index (index . 0) 1 -}}{{- end -}} {{- end -}} {{- with $val -}} {{- $clean := . | htmlUnescape -}} {{- if and (eq $field "og:image") (not $img) -}}{{- $img = $clean -}}{{- end -}} {{- if and (eq $field "og:title") (not $title) -}}{{- $title = $clean -}}{{- end -}} {{- if and (eq $field "og:description") (not $desc) -}}{{- $desc = $clean -}}{{- end -}} {{- if and (eq $field "og:site_name") (not $site) -}}{{- $site = $clean -}}{{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- if not $title -}}{{- $title = $url -}}{{- end -}} {{- if not $site -}}{{- $site = (urls.Parse $url).Host | replaceRE `^www\.` "" -}}{{- end -}} {{- /* one stylesheet per page, no matter how many cards */ -}} {{- if not (.Page.Scratch.Get "ddFlinkStyle") -}} {{- .Page.Scratch.Set "ddFlinkStyle" true -}} {{- end -}}
{{ $label }} {{- with $img }} {{- else }} {{- end }} {{ $site }} {{ $title }} {{- with $desc }}{{ . }}{{ end }} {{ $cta }}
